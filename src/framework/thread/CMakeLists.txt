cmake_minimum_required(VERSION 3.10.0)
project(QOR VERSION 0.1.0 LANGUAGES C CXX)

add_library(qor_thread SHARED qor_thread.cpp currentthread.cpp thread.cpp threadcontext.cpp)
find_package(Threads REQUIRED)
if(THREADS_HAVE_PTHREAD_ARG)
  target_compile_options(qor_thread PUBLIC "-pthread")
endif()
if(CMAKE_THREAD_LIBS_INIT)
  target_link_libraries(qor_thread "${CMAKE_THREAD_LIBS_INIT}")
endif()
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fno-threadsafe-statics -fvisibility=hidden")
target_compile_definitions(qor_thread PUBLIC -DQOR_THREAD)
target_link_libraries(qor_thread qor_module qor_host qor_datastructures qor_objectcontext qor_sync qor_injection)
